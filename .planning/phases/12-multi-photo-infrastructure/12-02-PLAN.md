

  ```typescript
  const sets: string[] = []
  const binds: any[] = []
  if (body.description !== undefined) { sets.push('description = ?'); binds.push(body.description) }
  if (body.location !== undefined) { sets.push('location = ?'); binds.push(body.location) }
  if (body.photo_keys !== undefined) {
    if (!Array.isArray(body.photo_keys) || body.photo_keys.length > 5) {
      return Response.json({ success: false, error: 'photo_keys는 0-5개 배열이어야 합니다' }, { status: 400 })

  if (sets.length === 0) {

  }
  binds.push(fid, scheduleItemId)
  await env.DB.prepare(`UPDATE legal_findings SET ${sets.join(', ')} WHERE id = ? AND schedule_item_id = ?`).bind(...binds).run()
  ```
