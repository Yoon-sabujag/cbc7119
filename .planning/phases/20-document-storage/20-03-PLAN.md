```typescript
import { jsonError, jsonOk, isDocType, type Env } from './_helpers'

// GET /api/documents?type=plan|drill[&year=YYYY]
// Auth: any authenticated staff (middleware enforces JWT; no role check)
// Response: { success: true, data: Array<{ id, type, year, title, filename, size, content_type, uploaded_at, uploaded_by_name }> }
export const onRequestGet: PagesFunction<Env> = async (ctx) => {
  const url = new URL(ctx.request.url)
  const type = url.searchParams.get('type')
  const yearRaw = url.searchParams.get('year')

  if (!isDocType(type)) {
    return jsonError(400, 'type은 plan 또는 drill이어야 합니다')
  }

  let year: number | null = null
  if (yearRaw !== null) {
    const y = Number(yearRaw)
    if (!Number.isInteger(y) || y < 2000 || y > 2100) {
      return jsonError(400, 'year가 유효하지 않습니다')
    }
    year = y
  }

  try {
    const sql = year === null
      ? `SELECT d.id, d.type, d.year, d.title, d.filename, d.size, d.content_type, d.uploaded_at, s.name AS uploaded_by_name
         FROM documents d
         LEFT JOIN staff s ON s.id = d.uploaded_by
         WHERE d.type = ? AND d.deleted_at IS NULL
         ORDER BY d.year DESC, d.uploaded_at DESC`
      : `SELECT d.id, d.type, d.year, d.title, d.filename, d.size, d.content_type, d.uploaded_at, s.name AS uploaded_by_name
         FROM documents d
         LEFT JOIN staff s ON s.id = d.uploaded_by
         WHERE d.type = ? AND d.year = ? AND d.deleted_at IS NULL
         ORDER BY d.uploaded_at DESC`

    const stmt = year === null
      ? ctx.env.DB.prepare(sql).bind(type)
      : ctx.env.DB.prepare(sql).bind(type, year)

    const { results } = await stmt.all()
    return jsonOk(results || [])
  } catch (e) {
    console.error('DOCUMENTS_LIST_FAILED', e)
    return jsonError(500, '문서 목록 조회에 실패했습니다')
  }
}
```

Create `cha-bio-safety/functions/api/documents/[id].ts` with EXACTLY this content:

```typescript
import { jsonError, type Env } from './_helpers'

// GET /api/documents/{id}
// Auth: any authenticated staff
// Response: binary stream with Content-Type, Content-Length, Content-Disposition
export const onRequestGet: PagesFunction<Env> = async (ctx) => {
  const idRaw = ctx.params.id as string
  const id = Number(idRaw)
  if (!Number.isInteger(id) || id <= 0) {
    return jsonError(400, '문서 id가 유효하지 않습니다')
  }

  try {
    const row = await ctx.env.DB.prepare(
      `SELECT id, filename, r2_key, size, content_type
       FROM documents
       WHERE id = ? AND deleted_at IS NULL`
    ).bind(id).first<{ id: number; filename: string; r2_key: string; size: number; content_type: string }>()

    if (!row) return jsonError(404, '문서를 찾을 수 없습니다')

    const obj = await ctx.env.STORAGE.get(row.r2_key)
    if (!obj) return jsonError(404, '파일을 찾을 수 없습니다')

    // D-14: UTF-8 encoded filename for Korean file name support
    const encodedName = encodeURIComponent(row.filename)
    const disposition = `attachment; filename*=UTF-8''${encodedName}`

    return new Response(obj.body, {
      headers: {
        'Content-Type':        row.content_type || 'application/octet-stream',
        'Content-Length':      String(row.size),
        'Content-Disposition': disposition,
        'Cache-Control':       'private, max-age=3600',
      },
    })
  } catch (e) {
    console.error('DOCUMENT_DOWNLOAD_FAILED', e)
    return jsonError(500, '문서 다운로드에 실패했습니다')
  }
}
```

Notes:
- `obj.body` is a ReadableStream piped directly into the Response — no buffering (per D-15, 130MB streams are network-bound and fit in CPU 30s limit).
- `filename*=UTF-8''` format is required for Korean filenames per D-14.
- Soft-deleted rows excluded via `deleted_at IS NULL`.

```javascript
(async () => {
  const token = JSON.parse(localStorage.getItem('auth-storage')).state.token
  const H = { 'Authorization': `Bearer ${token}` }

  // 12 MB synthetic PDF-ish blob (content doesn't have to be a real PDF for the smoke test,
  // but contentType must match an allowed MIME and filename must end in an allowed ext)
  const bytes = new Uint8Array(12 * 1024 * 1024)
  crypto.getRandomValues(bytes.subarray(0, 1024))  // seed random head
  const blob = new Blob([bytes], { type: 'application/pdf' })

  // 1. create
  const createRes = await fetch('/api/documents/multipart/create', {
    method: 'POST',
    headers: { ...H, 'Content-Type': 'application/json' },
    body: JSON.stringify({
      type: 'plan',
      year: 2026,
      title: 'Phase 20 smoke test',
      filename: 'smoke-test.pdf',
      contentType: 'application/pdf',
      size: blob.size,
    }),
  }).then(r => r.json())
  console.log('create →', createRes)
  if (!createRes.success) throw new Error('create failed')
  const { uploadId, key, partSize } = createRes.data

  // 2. upload-part ×2 (10 MB + 2 MB)
  const parts = []
  let pn = 1
  for (let off = 0; off < blob.size; off += partSize) {
    const chunk = blob.slice(off, off + partSize)
    const r = await fetch(`/api/documents/multipart/upload-part?uploadId=${encodeURIComponent(uploadId)}&key=${encodeURIComponent(key)}&partNumber=${pn}`, {
      method: 'PUT',
      headers: { ...H, 'Content-Type': 'application/octet-stream' },
      body: chunk,
    }).then(r => r.json())
    console.log(`part ${pn} →`, r)
    if (!r.success) throw new Error(`part ${pn} failed`)
    parts.push(r.data)
    pn++
  }

  // 3. complete
  const completeRes = await fetch('/api/documents/multipart/complete', {
    method: 'POST',
    headers: { ...H, 'Content-Type': 'application/json' },
    body: JSON.stringify({
      uploadId, key, parts,
      type: 'plan', year: 2026,
      title: 'Phase 20 smoke test',
      filename: 'smoke-test.pdf',
      size: blob.size,
      contentType: 'application/pdf',
    }),
  }).then(r => r.json())
  console.log('complete →', completeRes)
  if (!completeRes.success) throw new Error('complete failed')
  const docId = completeRes.data.id

  // 4. list
  const listRes = await fetch('/api/documents?type=plan&year=2026', { headers: H }).then(r => r.json())
  console.log('list →', listRes)
  const found = listRes.data.find(d => d.id === docId)
  if (!found) throw new Error('list does not contain new doc')
  console.log('found in list:', found)

  // 5. download (HEAD only — verify headers)
  const dl = await fetch(`/api/documents/${docId}`, { headers: H })
  console.log('download status:', dl.status)
  console.log('Content-Type:', dl.headers.get('Content-Type'))
  console.log('Content-Length:', dl.headers.get('Content-Length'))
  console.log('Content-Disposition:', dl.headers.get('Content-Disposition'))
  const buf = await dl.arrayBuffer()
  console.log('downloaded bytes:', buf.byteLength, 'expected:', blob.size)

  console.log('✅ SMOKE TEST PASSED')
})().catch(e => console.error('❌ SMOKE TEST FAILED', e))
```
