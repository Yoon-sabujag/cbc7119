   ```typescript
   export const remediationApi = {
     list: (params: { status?: string; category?: string; days?: number }) => {
       const q = new URLSearchParams()
       if (params.status && params.status !== 'all') q.set('status', params.status)
       if (params.category) q.set('category', params.category)
       if (params.days !== undefined) q.set('days', String(params.days))
       const qs = q.toString()
       return api.get<{ records: import('../types').RemediationRecord[]; categories: string[] }>(
         `/remediation${qs ? '?' + qs : ''}`
       )
     },
     get: (recordId: string) =>
       api.get<import('../types').RemediationRecord>(`/remediation/${recordId}`),
   }
   ```

Note: `days=0` means all-time. The list function passes `days` as-is to the query string; the API endpoint interprets `days=0` as no date filter.
