}

<!-- API client created by Plan 01 in src/utils/api.ts -->
export const elevatorInspectionApi = {
  getFindings: (elevatorId, inspectionId) => api.get<ElevatorInspectionFinding[]>(`/elevators/${elevatorId}/inspections/${inspectionId}/findings`),
  createFinding: (elevatorId, inspectionId, body) => api.post<{ id: string }>(`/elevators/${elevatorId}/inspections/${inspectionId}/findings`, body),
  resolveFinding: (elevatorId, inspectionId, fid, body) => api.post<void>(`/elevators/${elevatorId}/inspections/${inspectionId}/findings/${fid}/resolve`, body),
  getNextInspection: () => api.get<ElevatorNextInspection[]>('/elevators/next-inspection'),
}
