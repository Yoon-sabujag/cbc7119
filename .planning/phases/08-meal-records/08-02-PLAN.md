  if (providedMeals === 0) return
  const current = mealMap[ymd] ?? 0
  const next = (current + 1) > providedMeals ? 0 : current + 1
  // Optimistic update
  setOptimisticMealMap(prev => ({ ...prev, [ymd]: next }))
  try {
    await mealApi.upsert(ymd, next)
    queryClient.invalidateQueries({ queryKey: ['meal', year, month] })

  }
}, [year, month, mealMap])
